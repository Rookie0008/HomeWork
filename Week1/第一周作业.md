```python
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: ListNode) -> ListNode:
        if not head:return
        pre = None
        cur = head
        while cur:
            nxt = cur.next
            cur.next = pre
            pre = cur
            cur = nxt
        return pre
```

```python
class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        i, j = m - 1, n - 1
        k = m + n - 1
        for k in range(k, -1, -1):
            if j < 0 and i >= 0 or j >= 0 and i >= 0 and nums1[i] >= nums2[j]:
                nums1[k] = nums1[i]
                i -= 1
            else:
                nums1[k] = nums2[j]
                j -= 1
```

```python
    def hasCycle(self, head: ListNode) -> bool:
        if not head:return None
        fast = head
        slow = head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast == slow:return True
```

```python
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: ListNode) -> ListNode:
        fast, slow, flag = head, head, head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            if fast == slow:break
        else:return None
        while flag != slow:
            flag = flag.next
            slow = slow.next
        return flag
```

```python
class Solution:
    def isValid(self, s: str) -> bool:
        dic = {'}':'{', ']':'[', ')':'('}
        stack = []
        for s_i in s:
            if s_i in '([{':
                stack.append(s_i)
            elif stack and s_i and dic[s_i] == stack[-1]:
                stack.pop()
            else:return False
        return len(stack) == 0
```

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseKGroup(self, head: ListNode, k: int) -> ListNode:
        def reverse(head):
            pre, cur = None, head
            for i in range(k):
                temp = cur.next
                cur.next = pre
                pre = cur
                cur = temp
            head.next = cur
            return pre
        empty = ListNode()
        empty.next = head
        test, cur = empty, empty
        length = 0
        while test.next:
            length += 1
            test = test.next
        count = length // k
        for i in range(count):
            cur.next = reverse(cur.next)
            for i in range(k):
                cur = cur.next
        return empty.next
```

```python
class MinStack:

    def __init__(self):
        self.stack = []

    def push(self, val: int) -> None:
        if not self.stack:self.stack.append((val, val))
        else:
            min_num = self.stack[-1][1]
            if val > min_num:
                self.stack.append((val, min_num))
            else:self.stack.append((val, val))

    def pop(self) -> None:
        self.stack.pop()
        
    def top(self) -> int:
        return self.stack[-1][0]

    def getMin(self) -> int:
        return self.stack[-1][1]
```
